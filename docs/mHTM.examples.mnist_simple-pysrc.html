 74                  # Create the region

 77                  # Train the region 
 78                  sp.fit(x[tr], y[tr])

 80                  # Test the base classifier 
 81                  clf = LinearSVC(random_state=seed) 
 82                  clf.fit(x[tr], y[tr]) 
 83                  score = clf.score(x[te], y[te]) 
 84                  print 'SVM Only Accuracy: {0:.2f}%'.format(score * 100) 
 85                   
 86                  # Test the region for the column method 
 87                  score = sp.score(x[te], y[te]) 
 88                  print 'Column Accuracy: {0:.2f}%'.format(score * 100)

 90                  # Test the region for the probabilistic method 
 91                  score = sp.score(x[te], y[te], tr_x=x[tr], score_method='prob') 
 92                  print 'Probabilistic Accuracy: {0:.2f}%'.format(score * 100) 
 93                   
 94                  # Test the region for the dimensionality reduction method

 99                  print 'Reduction Accuracy: {0:.2f}%'.format(score * 100)
